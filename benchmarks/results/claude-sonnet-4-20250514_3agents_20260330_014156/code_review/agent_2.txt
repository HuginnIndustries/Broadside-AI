def get_users(db, filters: Optional[Dict[str, Any]] = None) -> List[Dict[str, Any]]:
    """
    Retrieve users from database with optional filters.
    
    Args:
        db: Database connection object
        filters: Optional dictionary of column:value filters
        
    Returns:
        List of user dictionaries (excluding passwords)
    """
    if filters is None:
        filters = {}
    
    # Validate column names against whitelist
    allowed_columns = {'name', 'email', 'status', 'created_at'}
    invalid_columns = set(filters.keys()) - allowed_columns
    if invalid_columns:
        raise ValueError(f"Invalid filter columns: {invalid_columns}")
    
    # Use parameterized query
    query = "SELECT id, name, email, status, created_at FROM users WHERE 1=1"
    params = []
    
    for column, value in filters.items():
        query += f" AND {column} = ?"
        params.append(value)
    
    try:
        results = db.execute(query, params)
        return [dict(row) for row in results]  # Don't include passwords
