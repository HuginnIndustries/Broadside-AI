def get_users(db, filters={}):
    """
    Retrieves users from the database based on provided filters.

    Args:
        db: The database connection object.
        filters: A dictionary of filter criteria.  (Example: {'username': 'john_doe', 'age': 30})

    Returns:
        A list of users.
    """
    query = f"SELECT * FROM users WHERE 1=1"
    for key, val in filters.items():
