from typing import Optional, Dict, Any, List, Generator
import re

def get_filtered_users(
    db, 
    filters: Optional[Dict[str, Any]] = None,
    columns: Optional[List[str]] = None
) -> Generator[Dict[str, Any], None, None]:
    """Retrieve users with optional filtering.
    
    Args:
        db: Database connection object
        filters: Dictionary of column=value filters
        columns: List of columns to select (None for all except password)
        
    Yields:
        User dictionaries without password field
    """
    if filters is None:
        filters = {}
    
    # Validate column names to prevent SQL injection
    valid_columns = {"id", "username", "email", "created_at"}  # Example valid columns
    for key in filters.keys():
        if key not in valid_columns:
            raise ValueError(f"Invalid filter column: {key}")
    
    # Build safe parameterized query
    query_parts = ["SELECT"]
    if columns:
        query_parts.append(", ".join(columns))
    else:
        # Explicitly list all columns except password
        query_parts.append("id, username, email, created_at")
    
    query_parts.append("FROM users")
    
    params = []
    if filters:
        where_clauses = []
        for key, val in filters.items():
            where_clauses.append(f"{key} = ?")
            params.append(val)
        query_parts.append("WHERE " + " AND ".join(where_clauses))
    
    query = " ".join(query_parts)
    
    try:
        results = db.execute(query, params)
        for row in results:
            user = dict(row)
            # DO NOT decrypt or include password in response
            if 'password' in user:
                del user['password']
            yield user
    except Exception as e:
        # Log the error appropriately
        raise DatabaseError(f"Failed to fetch users: {str(e)}")
